## 3. Обнаружение арбитражной возможности

```python
def find_arbitrage(bid_exchange1: float, ask_exchange2: float,
                   fee1: float = 0.001, fee2: float = 0.001) -> dict | None:
    """
    Ищет арбитраж: купить на exchange2 по ask, продать на exchange1 по bid.
    Возвращает None если возможности нет.
    """
    spread = bid_exchange1 - ask_exchange2
    costs = ask_exchange2 * (fee1 + fee2)
    net_profit_pct = (spread - costs) / ask_exchange2 * 100

    if net_profit_pct > float(os.getenv('MIN_SPREAD_PCT', 0.3)):
        return {
            'buy_at': ask_exchange2,
            'sell_at': bid_exchange1,
            'spread_pct': net_profit_pct,
        }
    return None
```

## 5. Скелет основного цикла бота

```python
import asyncio
from loguru import logger
from dotenv import load_dotenv

load_dotenv()
DRY_RUN = os.getenv('DRY_RUN', 'true').lower() == 'true'

async def run_bot():
    logger.info("Бот запущен. DRY_RUN={}", DRY_RUN)
    while True:
        try:
            # 1. Получить цены с двух бирж
            # 2. Найти арбитражную возможность
            # 3. Проверить риски (размер позиции, дневной лимит)
            # 4. Исполнить ордера (или пропустить если DRY_RUN)
            # 5. Записать в лог
            await asyncio.sleep(0.1)  # пауза между итерациями
        except Exception as e:
            logger.error("Ошибка в основном цикле: {}", e)
            await asyncio.sleep(5)

if __name__ == '__main__':
    asyncio.run(run_bot())
```
